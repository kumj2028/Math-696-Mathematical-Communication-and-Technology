    and the identity $\sin(x)^2 + \cos(x)^2 = 1$, what is the derivative of $\tan(x)$?
    \begin{enumerate}
      \item $\frac{\cos(x)}{\sin(x)}$
      \item $\frac{1}{\sin(x)^2}$
      \item $1$
      \item $\frac{1}{\cos(x)^2}$
      \item $0$
    \end{enumerate}

\end{problem}
